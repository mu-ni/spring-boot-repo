
# install vue-cli
~~~
npm install -g vue-cli
~~~

# init project
~~~
vue init webpack vue-icp
~~~

# run for debug
~~~
npm run dev
~~~

# build for production
~~~
npm run build
~~~

# deployment
~~~
1. copy content in /dist(/static & index.html)
2. paste to spring-boot project src\main\resources\public
3. run spring-boot project
~~~
